  5 6 7 8 6.5
  9 10 11 12 10.5
mean 5 6 7 8

mean(dim=0)-->[5., 6., 7., 8.]

mean(dim=1)-->[ 2.5000, 6.5000, 10.5000]

mean(dim=0,keepdim=True)-->[[5., 6., 7., 8.]]

mean(dim=1,keepdim=True)-->[[ 2.5000],[ 6.5000],[10.5000]]
